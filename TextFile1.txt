CREATE PROC PROC_THEMTHELOAI (@MATHELOAI CHAR(10), @TENTHELOAI NVARCHAR(50))
AS
BEGIN
INSERT INTO dbo.THELOAI
(
    MaTheLoai,
    TenTheLoai
)
VALUES
(   @MATHELOAI, -- MaTheLoai - char(10)
    @TENTHELOAI -- TenTheLoai - nvarchar(50)
    )
END
------------------------
CREATE PROC PROC_SUATHELOAI (@MATHELOAI CHAR(10), @TENTHELOAI NVARCHAR(50))
AS
BEGIN
UPDATE dbo.THELOAI SET TenTheLoai = @TENTHELOAI WHERE MaTheLoai = @MATHELOAI
END
-------------------
CREATE PROC PROC_XOATHELOAI (@MATHELOAI CHAR(10))
AS
BEGIN
DELETE dbo.THELOAI WHERE MaTheLoai = @MATHELOAI
END
